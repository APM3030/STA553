0.2 Create the Scatter plot

# upload penguin dataset
penguins <- "https://raw.githubusercontent.com/APM3030/STA553/main/data/w02-penguins(1).csv"
penguins <- read_csv(penguins)

-- Column specification ----------------------------------------------------------------------------
cols(
  X1 = col_double(),
  species = col_character(),
  island = col_character(),
  bill_length_mm = col_double(),
  bill_depth_mm = col_double(),
  flipper_length_mm = col_double(),
  body_mass_g = col_double(),
  sex = col_character(),
  year = col_double()
)
## rescale body_mass_g
body_mass_g_modified <- (penguins$body_mass_g)/2750
penguins <- cbind(penguins, body_mass_g_modified)

# upload image
myurl <- "https://github.com/APM3030/STA553/blob/66a77e123e304a54b2e80230d8d89c9aef1059af/IMAGES/PENGUIN.jpg?raw=true"
z <- tempfile()
download.file(myurl,z,mode="wb")
trying URL 'https://github.com/APM3030/STA553/blob/66a77e123e304a54b2e80230d8d89c9aef1059af/IMAGES/PENGUIN.jpg?raw=true'
Content type 'image/jpeg' length 235845 bytes (230 KB)
downloaded 230 KB
pic <- readJPEG(z)
raster.penguin <- as.raster(pic)

# create plot
bill.length = penguins$bill_length_mm
flipper.length = penguins$flipper_length_mm
penguin.mass = penguins$body_mass_g_modified
species = penguins$species
## identifying the ID of penguins
adelie.id = which(species=="Adelie")
gentoo.id = which(species=="Gentoo")
chinstrap.id = which(species=="Chinstrap")
## color code
col.code = c(alpha("red",0.5),alpha("blue",0.5),alpha("cyan",0.5))
## making an empty plot: type = "n" ==> no point
plot(bill.length, flipper.length, main = "bill legnth vs flipper length", type = "n")
## change the point size based on their body weight
points(bill.length[adelie.id], flipper.length[adelie.id], 
       pch = 19, col = col.code[1], cex = penguin.mass[adelie.id])
points(bill.length[gentoo.id], flipper.length[gentoo.id], 
       pch = 19, col = col.code[2], cex = penguin.mass[gentoo.id])
points(bill.length[chinstrap.id], flipper.length[chinstrap.id], 
       pch = 19, col = col.code[3], cex = penguin.mass[chinstrap.id])
legend("topleft", c("Adelie", "Gentoo", "Chinstrap"), 
                  col=col.code,
                  pch=c(19, 19, 19))
## code for image
rasterImage(raster.penguin,55,210,60,225)
text(52.5, 180, "The size of points is proportional to penguin mass", col = "purple", cex = 0.55)
